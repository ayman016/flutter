Then Principe of pushAndRemoveUntil work to remove all page and back to the main page like the first time :



MaterialButton(
              color: Colors.red,
              textColor: Colors.purple,
              onPressed: () {
                Navigator.of(
                  context,
                ).pushAndRemoveUntil(MaterialPageRoute(builder: (context) => Screenhome()),(Route) => false );
              },
              child: Text('back'),
            ),


And also add that to main for route :



Example :

import 'package:backdev/aboutus.dart';
import 'package:backdev/contactpage.dart';
import 'package:backdev/screenhome.dart';
import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(debugShowCheckedModeBanner: false, home: Screenhome(),
    routes: {
      'home ' : (context) => Screenhome(),
      'aboutus' : (context) => Aboutus(),
      'contact' : (context) => Contact(),
    },
    
    );
  }
}

 

MaterialButton(
              color: Colors.red,
              textColor: Colors.white,
              onPressed: () {
                Navigator.of(context).pushNamed('contact');
              },
              child: Text('go to contact screen'),
            ),
